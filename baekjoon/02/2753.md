import java.util.Scanner;

class Main {
    public static void main(String[] args){
        Scanner sc= new Scanner(System.in);
        int year = sc.nextInt();
        
        boolean leap_year = false;
        if(year%4==0){
            leap_year = true;
            if(year%100==0){
                leap_year = false;
                if(year%400==0){
                    leap_year = true;
                }
            }
        }
        
        if(leap_year) {
            System.out.println("1");
        } else if(!leap_year) {
            System.out.println("0");
        }
    }
}
