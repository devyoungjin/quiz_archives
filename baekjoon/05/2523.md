import java.util.Scanner;

class Main {
   public static void main(String[] args) {
      Scanner sc = new Scanner(System.in);

      int row = sc.nextInt();
      int col = (2*row)-1;

      for (int i = 0; i < col; i++) {
         if(i<row){
            for (int j = 0; j < i+1; j++) {
               System.out.print("*");
            }
         } else {
            for (int j = row-(i/2); j > 0; j--) {
               System.out.print("*");
            }
         }
         System.out.println("");
      }
   }
}
