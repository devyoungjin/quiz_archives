import java.util.Scanner;

class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int N = sc.nextInt();

        Integer[] numbers = new Integer[N];

        int max = -1000000;
        int min = 1000000;

        for (int i = 0; i < N; i++) {
            numbers[i] = sc.nextInt();
            if (numbers[i]>max){
                max = numbers[i];
            }
            if (numbers[i]<min){
                min = numbers[i];
            }
        }
        System.out.print(min+" "+max);
    }
}
