import java.util.Scanner;

public class Main {
    int factorial(int num){
        int result = 1;
        for (int i = num; i > 0; i--) {
            result *= i;
        }
        return result;
    }

    public static void main(String[] args) {
        Main m = new Main();
        Scanner sc = new Scanner(System.in);

        int num = sc.nextInt();
        int factorial = m.factorial(num);

        System.out.print(factorial);
    }
}
